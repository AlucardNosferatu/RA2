int INI_EX::ParseInteger ( const int  nDefault) [inline] 
                                             {
                int buffer = nDefault;

                const char *pValue = this->buffer();

                if(*pValue == '$') {
                        sscanf(pValue, "$%d", &buffer);
                        return buffer;
                }

                if(tolower(pValue[strlen(pValue) - 2]) == 'h') {
                        sscanf(pValue, "%xh", &buffer);
                        return buffer;
                }

                sscanf(pValue, "%d", &buffer);
                return buffer;
        }
